const wordlistName = document.querySelector('#wordlist-name')?.textContent.trim() || '';

const anchors = document.querySelectorAll('#myWordlist a[href]');
const hrefs = Array.from(anchors, a => a.getAttribute('href'));

const multilineText = [wordlistName, ...hrefs].join('\n');

// Copy to clipboard (works in most dev consoles)
copy(multilineText);
